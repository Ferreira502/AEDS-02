============================================================
MODELO DE QUESTAO - ESTRUTURA DOIDONA COM CPF E CNPJ
============================================================

SUPOSICAO ADOTADA
------------------------------------------------------------

O metodo recebe apenas o numero do documento como String.

- CPF possui 11 digitos.
- CNPJ possui 14 digitos.
- CPF segue: hash -> rehash -> arvore -> lista ordenada.
- CNPJ segue: hash -> lista encadeada ordenada.
- A arvore e ordenada por uma chave calculada a partir do CPF.
- Cada no da arvore aponta para uma lista de CPFs com a mesma chave.

Em uma prova, as funcoes hash devem ser trocadas pelas formulas exatas
fornecidas no enunciado. Aqui foram escolhidas formulas simples.

DESENHO DA ESTRUTURA
------------------------------------------------------------

CPF:

                    tabelaCPF
                 +------+------+------+
                 | CPF  | CPF  | null |
                 +------+------+------+
                    |       |
                 hash     rehash
                            |
             se as duas posicoes estiverem ocupadas
                            v
                        arvore BST
                       /          \
                 [chave 2]      [chave 8]
                     |               |
                     v               v
             lista ordenada   lista ordenada
             CPF -> CPF       CPF -> CPF

CNPJ:

                    tabelaCNPJ
                 +------+------+------+
                 |  *   |  *   |  *   |
                 +--|---+--|---+--|---+
                    v      v      v
                  lista  lista  lista
                 orden. orden. orden.

class Celula
{
    public String documento;
    public Celula prox;

    public Celula(String documento)
    {
        this.documento = documento;
        this.prox = null;
    }
}

class Lista
{
    private Celula primeiro;

    public Lista()
    {
        primeiro = null;
    }

    // Insere em ordem crescente e nao aceita repeticao.
    public void inserirOrdenado(String documento) throws Exception
    {
        Celula anterior = null;
        Celula atual = primeiro;

        while (atual != null && atual.documento.compareTo(documento) < 0)
        {
            anterior = atual;
            atual = atual.prox;
        }

        if (atual != null && atual.documento.equals(documento))
        {
            throw new Exception("Documento repetido");
        }

        Celula nova = new Celula(documento);
        nova.prox = atual;

        if (anterior == null)
        {
            primeiro = nova;
        }
        else
        {
            anterior.prox = nova;
        }
    }
}

class NoArvore
{
    public int chave;
    public Lista cpfs;
    public NoArvore esq, dir;

    public NoArvore(int chave, String cpf) throws Exception
    {
        this.chave = chave;
        this.cpfs = new Lista();
        this.cpfs.inserirOrdenado(cpf);
        this.esq = this.dir = null;
    }
}

class ArvoreCPF
{
    private NoArvore raiz;

    public void inserir(int chave, String cpf) throws Exception
    {
        raiz = inserir(chave, cpf, raiz);
    }

    private NoArvore inserir(int chave, String cpf, NoArvore i)
        throws Exception
    {
        if (i == null)
        {
            i = new NoArvore(chave, cpf);
        }
        else if (chave < i.chave)
        {
            i.esq = inserir(chave, cpf, i.esq);
        }
        else if (chave > i.chave)
        {
            i.dir = inserir(chave, cpf, i.dir);
        }
        else
        {
            // Mesma chave: guarda o CPF na lista do no.
            i.cpfs.inserirOrdenado(cpf);
        }

        return i;
    }
}

class HashCNPJ
{
    private Lista[] tabela;
    private int tamanho;

    public HashCNPJ(int tamanho)
    {
        this.tamanho = tamanho;
        tabela = new Lista[tamanho];

        for (int i = 0; i < tamanho; i++)
        {
            tabela[i] = new Lista();
        }
    }

    private int hash(String cnpj)
    {
        // Formula ilustrativa: usa os ultimos algarismos.
        long numero = Long.parseLong(cnpj);
        return (int)(numero % tamanho);
    }

    public void inserir(String cnpj) throws Exception
    {
        int pos = hash(cnpj);
        tabela[pos].inserirOrdenado(cnpj);
    }
}

class DoidonaDocumento
{
    private String[] tabelaCPF;
    private int tamanhoCPF;
    private ArvoreCPF arvoreCPF;
    private HashCNPJ hashCNPJ;

    public DoidonaDocumento(int tamanhoCPF, int tamanhoCNPJ)
    {
        this.tamanhoCPF = tamanhoCPF;
        this.tabelaCPF = new String[tamanhoCPF];
        this.arvoreCPF = new ArvoreCPF();
        this.hashCNPJ = new HashCNPJ(tamanhoCNPJ);
    }

    private int hashCPF(String cpf)
    {
        long numero = Long.parseLong(cpf);
        return (int)(numero % tamanhoCPF);
    }

    private int rehashCPF(String cpf)
    {
        return (hashCPF(cpf) + 1) % tamanhoCPF;
    }

    private int chaveArvore(String cpf)
    {
        // Formula ilustrativa para escolher o no da arvore.
        return Character.getNumericValue(cpf.charAt(0));
    }

    public void inserir(String documento) throws Exception
    {
        if (documento == null)
        {
            throw new Exception("Documento invalido");
        }

        documento = documento.replaceAll("[^0-9]", "");

        if (documento.length() == 11)
        {
            inserirCPF(documento);
        }
        else if (documento.length() == 14)
        {
            hashCNPJ.inserir(documento);
        }
        else
        {
            throw new Exception("Documento deve ser CPF ou CNPJ");
        }
    }

    private void inserirCPF(String cpf) throws Exception
    {
        int pos = hashCPF(cpf);

        if (tabelaCPF[pos] == null)
        {
            tabelaCPF[pos] = cpf;
        }
        else if (tabelaCPF[pos].equals(cpf))
        {
            throw new Exception("CPF repetido");
        }
        else
        {
            pos = rehashCPF(cpf);

            if (tabelaCPF[pos] == null)
            {
                tabelaCPF[pos] = cpf;
            }
            else if (tabelaCPF[pos].equals(cpf))
            {
                throw new Exception("CPF repetido");
            }
            else
            {
                int chave = chaveArvore(cpf);
                arvoreCPF.inserir(chave, cpf);
            }
        }
    }
}

============================================================
EXPLICACAO BREVE
============================================================

1. O metodo inserir remove pontos, barras e tracos do documento.

2. Se houver 11 digitos, o documento e tratado como CPF:
   a) tenta a posicao hash;
   b) se houver colisao, tenta a posicao rehash;
   c) se houver nova colisao, insere na arvore;
   d) a arvore localiza o no pela chave;
   e) o CPF e colocado ordenadamente na lista desse no.

3. Se houver 14 digitos, o documento e tratado como CNPJ:
   a) calcula a posicao da hash;
   b) insere o CNPJ ordenadamente na lista daquela posicao.

4. A lista ordenada procura o primeiro elemento maior ou igual ao novo.
   O novo elemento e ligado antes dele. Assim, a lista permanece ordenada.

5. A comparacao entre documentos usa compareTo. Como todos os CPFs de uma
   lista possuem 11 digitos e todos os CNPJs possuem 14, a ordem textual e
   igual a ordem numerica e ainda preserva zeros no inicio.

6. O codigo verifica repeticoes na tabela e nas listas.
